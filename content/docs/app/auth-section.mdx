## Installation

```sh
bun add @olwiba/ui
```

## Usage

```tsx
import { AuthSection } from "@olwiba/ui";

// Centered (default)
<AuthSection
  onSubmit={(e) => { e.preventDefault(); signIn() }}
  onSso={() => signInWithSso()}
  signUpHref="/register"
  forgotPasswordHref="/forgot-password"
/>

// Split layout with decorative panel
<AuthSection
  layout="split"
  onSubmit={(e) => { e.preventDefault(); signIn() }}
/>

// Custom form
<AuthSection layout="split">
  <MyCustomForm />
</AuthSection>
```
