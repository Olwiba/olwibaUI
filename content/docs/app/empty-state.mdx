## Installation

```sh
bun add @olwiba/ui
```

## Usage

```tsx
import { EmptyState } from "@olwiba/ui";
import { FolderOpen } from "lucide-react";

<EmptyState
  icon={FolderOpen}
  title="No projects yet"
  description="Create your first project to get started."
  action={{ label: "New project", onClick: () => router.push("/new") }}
/>
```
