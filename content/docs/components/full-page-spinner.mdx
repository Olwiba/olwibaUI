## Installation

```sh
bun add @olwiba/ui
```

## Usage

```tsx
import { FullPageSpinner } from "@olwiba/ui";
```

```tsx
<FullPageSpinner />
<FullPageSpinner message="Loading your workspace..." />
```

Use as a route-level suspense fallback:

```tsx
import { Suspense } from "react";

<Suspense fallback={<FullPageSpinner message="Loading..." />}>
  <Routes />
</Suspense>
```
