## Installation

```sh
bun add @olwiba/ui
```

## Usage

```tsx
import { Suspensed } from "@olwiba/ui";
import { Spinner } from "@olwiba/cn";
```

```tsx
// Default fallback is <Spinner />
<Suspensed>
  <LazyComponent />
</Suspensed>

// Custom fallback
<Suspensed fallback={<Spinner className="size-4" />}>
  <LazyComponent />
</Suspensed>
```
