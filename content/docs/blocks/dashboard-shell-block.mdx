## Installation

```sh
bun add @olwiba/ui
```

## Usage

```tsx
import { DashboardShellBlock } from "@olwiba/ui";
import { LayoutDashboard, Plus } from "lucide-react";
import { Link } from "your-router"; // TanStack, Next.js, etc.

export function AppLayout({ children }) {
  return (
    <DashboardShellBlock
      brand={{ name: "My App", href: "/" }}
      navItems={[
        { icon: LayoutDashboard, label: "Dashboard", href: "/dashboard", isActive: true },
      ]}
      action={{ icon: Plus, label: "New item", href: "/new" }}
      user={{
        email: "user@example.com",
        plan: "free",
        onSignOut: () => signOut(),
      }}
      pageTitle="Dashboard"
      renderLink={({ href, children }) => <Link to={href}>{children}</Link>}
    >
      {children}
    </DashboardShellBlock>
  );
}
```

The `demo` prop wraps the shell in a rounded preview card — use it in docs sandboxes or storybook:

```tsx
<DashboardShellBlock demo />
```
