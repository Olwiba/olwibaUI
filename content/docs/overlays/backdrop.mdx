## Installation

```sh
bun add @olwiba/ui
```

## Usage

```tsx
import { Backdrop } from "@olwiba/ui";

// Gradient variants — wrap content
<div className="relative overflow-hidden">
  <Backdrop variant="mesh" animated />
  <Backdrop variant="radial" />
  <Backdrop variant="aurora" animated />
  {children}
</div>

// Grid variants — SVG overlay, position absolute
<div className="relative">
  <Backdrop variant="dots" size={24} />
  <Backdrop variant="lines" />
  <Backdrop variant="cross" />
</div>

// Noise texture overlay
<Backdrop variant="noise" opacity={0.04} blendMode="overlay" />

// Glow — follows cursor when followCursor is true
<Backdrop variant="glow" size={400} followCursor />
<Backdrop variant="glow" color="hsl(var(--primary) / 0.4)" blur="xl" />
```
